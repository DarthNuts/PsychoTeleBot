    InMemoryTicketRepository
)

    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',

)

    def __init__(self, token: str):
        """

        ticket_repo = InMemoryTicketRepository()

            session_repo=session_repo,
            ticket_repo=ticket_repo,
            state_machine=state_machine
        )

        await update.message.reply_text(response)
        
        logger.info(f"User {user_id} started the bot")

    async def handle_menu(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """–û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /menu"""
        user_id = str(update.effective_user.id)
        response = self.bot_service.process_message(user_id, "/menu")

        """–û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /clear"""
        user_id = str(update.effective_user.id)
        response = self.bot_service.process_message(user_id, "/clear")

    async def handle_help(self, update: Update, context: ContextTypes.DEFAULT_TYPE):

        help_text = """

–ë–æ—Ç –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç:
‚Ä¢ –ö–æ–Ω—Å—É–ª—å—Ç–∞—Ü–∏–∏ —Å–æ —Å–ø–µ—Ü–∏–∞–ª–∏—Å—Ç–∞–º–∏
‚Ä¢ –ö–æ–Ω—Å—É–ª—å—Ç–∞—Ü–∏–∏ —Å –ò–ò-–∞—Å—Å–∏—Å—Ç–µ–Ω—Ç–æ–º

"""
        await update.message.reply_text(help_text, parse_mode='Markdown')

    async def handle_message(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """–û–±—Ä–∞–±–æ—Ç–∫–∞ —Ç–µ–∫—Å—Ç–æ–≤—ã—Ö —Å–æ–æ–±—â–µ–Ω–∏–π"""

        """–ó–∞–ø—É—Å–∫ –±–æ—Ç–∞"""

        self.application.run_polling(allowed_updates=Update.ALL_TYPES)
